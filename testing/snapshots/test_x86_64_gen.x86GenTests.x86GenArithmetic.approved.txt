default rel
global _start
section .text
_start:
	push rbp
	mov rbp, rsp
	push r12
	push r13
	push r14
	push r15
	push rbx
	mov r10, 10
	mov QWORD[global_vars + 8], r10
	mov r11, 5
	mov r12, QWORD[global_vars + 8]
	add r12, r11
	mov r13, 2
	mov r14, r12
	imul r14, r13
	mov QWORD[global_vars + 16], r14
	mov r15, 2
	; --- DIV start ---
	push rax
	push rdx
	mov rax, QWORD[global_vars + 8]
	cqo
	idiv r15
	mov rbx, rax
	pop rdx
	pop rax
	; --- DIV end ---
	mov r10, QWORD[global_vars + 16]
	sub r10, rbx
	mov r11, QWORD[global_vars + 8]
	neg r11
	mov r12, r10
	add r12, r11
	mov QWORD[global_vars + 24], r12
	mov QWORD[global_vars + 32], 0
	mov r13, 100
	mov r14, QWORD[global_vars + 24]
	add r14, r13
	mov QWORD[global_vars + 32], r14
	mov rdi, QWORD[global_vars + 32]
	call print_int
	mov r15, LC0
	mov rdi, r15
	call print_string
	pop rbx
	pop r15
	pop r14
	pop r13
	pop r12
	mov rsp, rbp ; restore stack
	pop rbp
	xor rdi, rdi
	call exit

section .data
LC0:
	db 10, 0
section .bss
	global_vars resb 32
