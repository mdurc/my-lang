extern exit, string_length, print_string, print_char
extern print_newline, print_uint, print_int
extern read_char, read_word, parse_uint
extern parse_int, string_equals, string_copy
extern memcpy, malloc, free, clrscr

default rel
global _start
section .text
_start:
	push rbp
	mov rbp, rsp
	mov DWORD[global_vars + 4], 0
	sub rsp, 64
	mov rdi, rsp
	call read_word
	mov r10, rax
	push r10 ; saving caller-saved register r10
	call parse_int
	pop r10 ; restoring caller-saved register
	mov r10, rax
	mov DWORD[global_vars + 4], r10d
	mov QWORD[global_vars + 12], 0
	push r10 ; saving caller-saved register r10
	sub rsp, 64
	mov rdi, rsp
	call read_word
	pop r10 ; restoring caller-saved register
	mov r11, rax
	mov QWORD[global_vars + 12], r11
	mov rsp, rbp ; restore stack
	pop rbp
	xor rdi, rdi
	call exit

section .data
section .bss
	global_vars resb 16
